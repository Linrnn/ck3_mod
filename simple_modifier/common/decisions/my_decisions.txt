杀死无地者 = {
	decision_group_type = my_decision_group_types
	ai_check_interval = 0
	effect = {
		every_living_character = {
			if = {
				limit = {
					OR = { 
						has_variable = flavourise_camp_purpose_legitimists
						has_variable = flavourise_camp_purpose_brigands
						has_variable = flavourise_camp_purpose_explorers
						has_variable = flavourise_camp_purpose_scholars
						has_variable = flavourise_camp_purpose_wanderers
						has_variable = flavourise_camp_purpose_mercenaries
						is_landless_administrative = yes
					}
					my_trigger_kill_character = yes
					NOT = { is_independent_ruler = yes }
				}
				death = { death_reason = magic }
			}
		}
		every_close_or_extended_family_member = {
			if = {
				limit = {
					my_trigger_kill_character = yes
					NOT = { is_ruler = yes }
				}
				death = { death_reason = magic }
			}
		}
		every_courtier = {
			if = {
				limit = {
					my_trigger_kill_character = yes
					NOT = { is_ruler = yes }
				}
				death = { death_reason = magic }
			}
		}
		every_vassal_or_below = {
			if = {
				limit = {
					my_trigger_kill_character = yes
					NOT = { is_ruler = yes }
				}
				death = { death_reason = magic }
			}
		}
	}
	ai_potential = {
		always = no
	}
	ai_will_do = {
		base = 0
	}
}
